587 //---------------------------------------------------------------------
588 bool hexStringToByteArray( char *str,
589  uint32_t strlen,
590  uint8_t bytes[])
591 {
592  if ( strlen % 2 )
593  {
594 #ifdef DEBUG_SERIAL
595  Serial.print("ERROR: invalid string length to convert: ");
596  Serial.println(strlen);
597 #endif
598  return true;
599  }
600 
601  for ( uint32_t c = 0; c < (strlen / 2); c++ )
602  {
603  if ( str[2 * c] >= '0' && str[2 * c] <= '9' )
604  bytes[c] += (str[2 * c] - '0') << 4;
605  else if ( str[2 * c] >= 'a' && str[2 * c] <= 'f' )
606  bytes[c] += (str[2 * c] - 'a' + 10) << 4;
607  else if ( str[2 * c] >= 'A' && str[2 * c] <= 'F' )
608  bytes[c] += (str[2 * c] - 'A' + 10) << 4;
609  else
610  {
611 #ifdef DEBUG_SERIAL
612  Serial.print("ERROR: invalid hex string");
613 #endif
614  return true;
615  }
616 
617  if ( str[2 * c + 1] >= '0' && str[2 * c + 1] <= '9' )
618  bytes[c] += str[2 * c + 1] - '0';
619  else if ( str[2 * c + 1] >= 'a' && str[2 * c + 1] <= 'f' )
620  bytes[c] += str[2 * c + 1] - 'a' + 10;
621  else if ( str[2 * c + 1] >= 'A' && str[2 * c + 1] <= 'F' )
622  bytes[c] += str[2 * c + 1] - 'A' + 10;
623  else
624  {
625 #ifdef DEBUG_SERIAL
626  Serial.println("ERROR: invalid hex string");
627 #endif
628  return true;
629  }
630  }
631  return false;
